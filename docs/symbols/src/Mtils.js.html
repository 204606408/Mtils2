117 		 * @author Rui.Zhang
118 		 * @description   判断数据是否为网址(URL)
119 		 * @param {String} str_data 待校验的数据
120 		 * @returns {Boolean}, true:是网址
121 		 **/
122 		isUrl : function (str_data) {
123 			str_data = str_data || String(this);
124 		    var regular = /^\b(((https?|ftp):\/\/)?[-a-z0-9]+(\.[-a-z0-9]+)*\.(?:com|edu|gov|int|mil|net|org|biz|info|name|museum|asia|coop|aero|[a-z][a-z]|((25[0-5])|(2[0-4]\d)|(1\d\d)|([1-9]\d)|\d))\b(\/[-a-z0-9_:\@&?=+,.!\/~%\$]*)?)$/i
125 		    if (regular.test(str_data)) {

145 		/**
146 		 * @author Rui.Zhang
147 		 * @description 判断数据是否包含特殊字符, 包含中文标点
148 		 * @param {String} str_data 待校验的数据
149 		 * @returns {Boolean}, true:包含特殊字符, null表示传入参数为null 或者传入参数为空串
150 		 **/
151 		isContainsSpecialChar : function (str_data) {
152 		    if (str_data) {
153 		        //英文符号
154 		        var containSpecialForEnglish = RegExp(/[(\s)(\`)(\~)(\!)(\@)(\#)(\$)(\%)(\^)(\&)(\*)(\()(\))(\-)(\_)(\+)(\=)(\[)(\])(\{)(\})(\|)(\\)(\;)(\:)(\')(\")(\,)(\.)(\/)(\<)(\>)(\?)(\)]+/);
155 		        //中文符号
156 		        var containSpecialForChinese = RegExp(/[(\·)(\~)(\！)(\￥)(\%)(\……)(\&)(\*)(\（)(\）)(\——)(\【)(\】)(\；)(\：)(\”)(\“)(\’)(\，)(\《)(\。)(\》)(\？)(\、)(\‘)(\’)]+/);
157 
158 		        str_data = str_data || String(this);
159 		        return (containSpecialForEnglish.test(str_data) || containSpecialForChinese.test(str_data));
160 		    }
161 		    return null;
162 		},

188 		    } else {
