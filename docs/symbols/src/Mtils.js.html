253 		/**
254 		 * @author Rui.Zhang
255 		 * @description 判断是否为银行卡号
256 		 * @param {String} str_cardNo 待校验的数据
257 		 * @returns {Boolean}, true:是银行卡号
258 		 **/
259 		isBankCard : function (str_cardNo) {
260 			str_cardNo = str_cardNo || String(this);
261 		    if ("" == str_cardNo.trim() || undefined == str_cardNo) {
262 		        return false;
263 		    }
264 		    var lastNum = str_cardNo.substr(str_cardNo.length - 1, 1);//取出最后一位（与luhm进行比较）
265 
266 		    var first15Num = str_cardNo.substr(0, str_cardNo.length - 1);//前15或18位
267 		    var newArr=new Array();
268 		    for(var i=first15Num.length-1;i>-1;i--){    //前15或18位倒序存进数组
269 		        newArr.push(first15Num.substr(i,1));
270 		    }
271 		    var arrJiShu=new Array();  //奇数位*2的积 <9
272 		    var arrJiShu2=new Array(); //奇数位*2的积 >9
273 
274 		    var arrOuShu=new Array();  //偶数位数组
275 		    for(var j=0;j<newArr.length;j++){
276 		        if((j+1)%2==1){//奇数位
277 		            if(parseInt(newArr[j])*2<9)
278 		                arrJiShu.push(parseInt(newArr[j])*2);
279 		            else
280 		                arrJiShu2.push(parseInt(newArr[j])*2);
281 		        }
282 		        else //偶数位
283 		            arrOuShu.push(newArr[j]);
284 		    }
285 
286 		    var jishu_child1=new Array();//奇数位*2 >9 的分割之后的数组个位数
287 		    var jishu_child2=new Array();//奇数位*2 >9 的分割之后的数组十位数
288 		    for(var h=0;h<arrJiShu2.length;h++){
289 		        jishu_child1.push(parseInt(arrJiShu2[h])%10);
290 		        jishu_child2.push(parseInt(arrJiShu2[h])/10);
291 		    }
292 
293 		    var sumJiShu=0; //奇数位*2 < 9 的数组之和
294 		    var sumOuShu=0; //偶数位数组之和
295 		    var sumJiShuChild1=0; //奇数位*2 >9 的分割之后的数组个位数之和
296 		    var sumJiShuChild2=0; //奇数位*2 >9 的分割之后的数组十位数之和
297 		    var sumTotal=0;
298 		    for(var m=0;m<arrJiShu.length;m++){
299 		        sumJiShu=sumJiShu+parseInt(arrJiShu[m]);
300 		    }
301 
302 		    for(var n=0;n<arrOuShu.length;n++){
303 		        sumOuShu=sumOuShu+parseInt(arrOuShu[n]);
304 		    }
305 
306 		    for(var p=0;p<jishu_child1.length;p++){
307 		        sumJiShuChild1=sumJiShuChild1+parseInt(jishu_child1[p]);
308 		        sumJiShuChild2=sumJiShuChild2+parseInt(jishu_child2[p]);
309 		    }
310 		    //计算总和
311 		    sumTotal=parseInt(sumJiShu)+parseInt(sumOuShu)+parseInt(sumJiShuChild1)+parseInt(sumJiShuChild2);
312 
313 		    //计算Luhm值
314 		    var k= parseInt(sumTotal)%10==0?10:parseInt(sumTotal)%10;
315 		    var luhm= 10-k;
316 
317 		    if(lastNum==luhm){
318 		        return true;
319 		    }
320 		    else{
321 		        return false;
322 		    }
323 		},

326 		/**
327 		 * @author Rui.Zhang
328 		 * @description 判断是否为身份证号码
329 		 * @param {String} str_idCard 待校验的数据
330 		 * @param {String} sex 同时校验性别,不加则不校验性别,此值只可能 constant.WOMAN(0) 或者 constant.MAN(1)
331 		 * @returns {Boolean}, true:是身份证
332 		 **/
333 		isIdCard : function (str_idCard, sex) {
334 			str_idCard = str_idCard || String(this);
335 	        var check = function() {
336 	            var factorArr = new Array(7, 9, 10, 5, 8, 4, 2, 1, 6, 3, 7, 9, 10, 5, 8, 4, 2, 1);
337 	            var parityBit = new Array("1", "0", "X", "9", "8", "7", "6", "5", "4", "3", "2");
338 	            var varArray = new Array();
339 	            var lngProduct = 0;
340 	            var intCheckDigit;
341 	            var intStrLen = str_idCard.length;
342 	            var idNumber = str_idCard;
343 	            // initialize
344 	            if ((intStrLen != 15) && (intStrLen != 18)) {
345 	                return false;
346 	            }
347 	            // check and set value
348 	            for (i = 0; i < intStrLen; i++) {
349 	                varArray[i] = idNumber.charAt(i);
350 	                if ((varArray[i] < '0' || varArray[i] > '9') && (i != 17)) {
351 	                    return false;
352 	                } else if (i < 17) {
353 	                    varArray[i] = varArray[i] * factorArr[i];
354 	                }
355 	            }
356 
357 	            if (intStrLen == 18) {
358 	                //check date
359 	                var date8 = idNumber.substring(6, 14);
360 
361 	                if (!/^[0-9]{8}$/.test(date8)) {
362 	                    return false;
363 	                }
364 	                var year, month, day;
365 	                year = date8.substring(0, 4);
366 	                month = date8.substring(4, 6);
367 	                day = date8.substring(6, 8);
368 	                var iaMonthDays = [31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31]
369 	                if (year < 1700 || year > 2500) return false;
370 	                if (((year % 4 == 0) && (year % 100 != 0)) || (year % 400 == 0)) iaMonthDays[1] = 29;
371 	                if (month < 1 || month > 12) return false;
372 	                if (day < 1 || day > iaMonthDays[month - 1]) return false;
373 
374 	                // calculate the sum of the products
375 	                for (i = 0; i < 17; i++) {
376 	                    lngProduct = lngProduct + varArray[i];
377 	                }
378 	                // calculate the check digit
379 	                intCheckDigit = parityBit[lngProduct % 11];
380 	                // check last digit
381 	                if (varArray[17] != intCheckDigit) {
382 	                    return false;
383 	                }
384 	            }
385 	            //length is 15
386 	            else {
387 	                //check date
388 	                var date6 = idNumber.substring(6, 12);
389 
390 	                if (!/^[0-9]{6}$/.test(date6)) {
391 	                    return false;
392 	                }
393 	                var  month, day, year;
394 	                year = date6.substring(0, 2);	
395 	                month = date6.substring(2, 4);
396 	                day = date6.substring(4, 6);
397 	                if (!/^\d{2}$/.test(year)) return false;
398 	                if (month < 1 || month > 12) return false;
399 	                if (day < 1 || day > 31) return false;
400 	            }
401 	            return true;
402 	        }
403 	        if(str_idCard && check(str_idCard)) {
404 	            if(undefined != sex) {
405 	                var sexStr = undefined, tmp = 0;
406 	                if (15 == str_idCard.length) {
407 	                    tmp = str_idCard.substring(str_idCard.length - 1, str_idCard.length);
408 	                } else if (18 == str_idCard.length) {
409 	                    tmp = str_idCard.substr(str_idCard.length - 2, 1);
410 	                }
411 	                if (0 == tmp % 2) {
412 	                    sexStr = 0;
413 	                } else {
414 	                    sexStr = 1;
415 	                }
416 
417 	                if (sex != sexStr) return false;
418 	            }
419 	            return true;
420 	        }
421 	        return false;
422 	    }

445 		 * @description 获取字符串的字节长度

447 		 * @returns {Long}, 字符串长度 
448 		 **/
449 		getByteLength : function(str_data) {
450 			str_data = str_data || String(this);
451 			if(Mtils.validate.isString(str_data)) {
452 				var Zhlength = 0;// 全角
453 		        var Enlength = 0;// 半角
454 		        if(str_data) {
455 		            for (var i = 0; i < str_data.length; i++) {
456 		                if (str_data.substring(i, i + 1).match(/[^\x00-\xff]/ig) != null)
457 		                    Zhlength += 1;
458 		                else
459 		                    Enlength += 1;
460 		            }
461 		        }
462 		        // 返回当前字符串字节长度
463 		        return  (Zhlength * 2) + Enlength; //所占字节数
464 			}
465 			return 0;
466 		}
